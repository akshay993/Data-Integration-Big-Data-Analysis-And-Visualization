commands:

#Go to Desktop
cd Desktop


#Start Hadoop
start-hadoop.sh

#create input directory in the hadoop file system
hdfs dfs -mkdir -p akshay1

#Put the desired text file into the input folder by specifying the location of the text file(assuming the text file is on desktop)
hdfs dfs -put /home/hadoop/Desktop/word.txt  input

#Viewing content of input folder
hdfs dfs -ls input

#Creating executable of python file (assuiming the python files are on the desktop)
chmod +x *.py

#Running mapreduce
hadoop jar /home/hadoop/hadoop/share/hadoop/tools/lib/hadoop-streaming-2.6.4.jar -file ./mapper.py -mapper ./mapper.py -file ./reducer.py -reducer ./reducer.py -input input -output output

#View the output
hdfs dfs -cat output/*


##### How to set Python Path in Ubuntu ##########

nano .bashrc

export PYTHONPATH="${PYTHONPATH}:/usr/local/lib/python3.4/dist-packages"

export PATH="/usr/local/lib/python3.4/dist-packages:$PATH"
alias python=python3

#Exit .bashrc

#Now, go to  cd /home/hadoop/hadoop/etc/hadoop
#open hadoop-env.sh

nano hadoop-env.sh

#Add the following line
export PYTHONPATH=/usr/local/lib/python3.4/dist-packages

#Exit hadoop-env.sh
